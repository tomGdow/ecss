function makeRequest(url, fn) {
    let oReq = new XMLHttpRequest();
    oReq.addEventListener("load", fn);
    oReq.open('GET', url);
    oReq.send();
}

function addQueryStringToUrl(url, ...args) {
    var tmp = url + '?';
    if (args.length < 3) {
        // console.log(args);
        tmp = `${tmp}${args[0]}=${args[1]}`;
    }
    else {
        tmp = `${tmp}${args[0]}=${args[1]}`;
        for (let i = 2; i < args.length; i += 2) {
            tmp += `&${args[i]}=${args[i + 1]}`;
        }
    }
    return tmp;
}
